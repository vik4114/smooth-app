@override
void didChangeDependencies() {
  super.didChangeDependencies();

  if (mounted) {
    _model = context.watch<ContinuousScanModel>();
    _userPreferences = context.watch<UserPreferences>();
  }

  // Relaunch the feed after a hot reload
  if (_controller == null) {
    _startLiveFeed();
  }
}
